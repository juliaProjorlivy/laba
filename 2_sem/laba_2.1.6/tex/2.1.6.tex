\section*{–¢–µ–æ—Ä–µ—Ç–∏—á–µ—Å–∫–∏–µ —Å–≤–µ–¥–µ–Ω–∏—è}

\indent{–≠—Ñ—Ñ–µ–∫—Ç–æ–º –î–∂–æ—É–ª—è‚Äì–¢–æ–º—Å–æ–Ω–∞ –Ω–∞–∑—ã–≤–∞–µ—Ç—Å—è –∏–∑–º–µ–Ω–µ–Ω–∏–µ —Ç–µ–º–ø–µ—Ä–∞—Ç—É—Ä—ã –≥–∞–∑–∞, –º–µ–¥–ª–µ–Ω–Ω–æ
–ø—Ä–æ—Å–∞—á–∏–≤–∞—é—â–µ–≥–æ—Å—è –∏–∑ –æ–±–ª–∞—Å—Ç–∏ –≤—ã—Å–æ–∫–æ–≥–æ –≤ –æ–±–ª–∞—Å—Ç—å –Ω–∏–∑–∫–æ–≥–æ –¥–∞–≤–ª–µ–Ω–∏—è –≤ —É—Å–ª–æ–≤–∏—è—Ö 
—Ç–µ–ø–ª–æ–≤–æ–π –∏–∑–æ–ª—è—Ü–∏–∏. –ö–æ—ç—Ñ—Ñ–∏—Ü–∏–µ–Ω—Ç–æ–º –î–∂–æ—É–ª—è-–¢–æ–º—Å–æ–Ω–∞ –Ω–∞–∑—ã–≤–∞–µ—Ç—Å—è –≤–µ–ª–∏—á–∏–Ω–∞:}
\begin{equation}
    \mu_{\text{–¥-—Ç}} = \frac{\Delta T}{\Delta P} 
\end{equation} 
–í—ã–≤–µ–¥–µ–º –Ω–µ–∫–æ—Ç–æ—Ä—ã–µ —Ç–µ–æ—Ä–µ—Ç–∏—á–µ—Å–∫–∏–µ —Å–æ–æ—Ç–Ω–æ—à–µ–Ω–∏—è:
$$Q=0 \rightarrow \delta A = -\Delta U \rightarrow P_1 V_1-P_2 V_2 = 
(U_2 + \frac{\mu {\upsilon_2}^2}{2})-(U_1 + \frac{\mu {\upsilon_1}^2}{2})$$\\
–ú–æ–ª—è—Ä–Ω–∞—è —ç–Ω—Ç–∞–ª—å–ø–∏—è: $H = U+PV$, —Ç–æ–≥–¥–∞ 
$$H_1 - H_2 = \frac{\mu ({\upsilon_2}^2 - {\upsilon_1}^2)}{2}$$ 
–û–¥–Ω–∞–∫–æ –∫–∏–Ω–µ—Ç–∏—á–µ—Å–∫–∞—è —ç–Ω–µ—Ä–≥–∏—è –≥–∞–∑–∞ –æ–∫–∞–∑—ã–≤–∞–µ—Ç—Å—è –ø—Ä–µ–Ω–µ–±—Ä–µ–∂–∏–º–æ –º–∞–ª–æ–π, –ø–æ—ç—Ç–æ–º—É
$$H_1 \approx H_2$$
–†–∞—Å—Å–º–æ—Ç—Ä–∏–º —Ä–µ–∞–ª—å–Ω—ã–π –≥–∞–∑ –∫–∞–∫ –≥–∞–∑ –í–∞–Ω-–¥–µ—Ä-–í–∞–∞–ª—å—Å–∞:
$$\left ( P+\frac{a}{V^2}\right )(V-b) = RT$$
$$U = C_\text{v}T-\frac{a}{V}$$
–¢–æ–≥–¥–∞ —ç–Ω—Ç–∞–ª—å–ø–∏—è:
\begin{equation} \label{eq:entalp}
    H = U+PV=C_{\text{v}}T+RT\frac{V}{V-a} - \frac{2a}{V}
\end{equation}
–ü—Ä–∏ —ç—Ç–æ–º –¥–ª—è —É–ø—Ä–æ—â–µ–Ω–∏—è –≤—ã—á–∏—Å–ª–µ–Ω–∏–π –±—É–¥–µ–º —Å—á–∏—Ç–∞—Ç—å –≥–∞–∑ —Ä–∞–∑—Ä—è–∂–µ–Ω–Ω—ã–º –∏ 
—Å—á–∏—Ç–∞—Ç—å –æ–±—ä–µ–º –ø–æ —Ñ–æ—Ä–º—É–ª–µ –ö–ª–∞–π–ø–µ—Ä–æ–Ω–∞-–ú–µ–Ω–¥–µ–ª–µ–µ–≤–∞ $V \approx \frac{RT}{P}$, 
–∞ —Ç–∞–∫–∂–µ —Å —É—á–µ—Ç–æ–º —Ç–æ–≥–æ, —á—Ç–æ $b \ll V$ –∏–∑ \ref{eq:entalp} –ø–æ–ª—É—á–∞–µ–º:
\begin{equation} \label{eq:entalpy}
H \approx C_{\text{P}}T + P \left (b- \frac{2a}{RT} \right )
\end{equation}
–¢–∞–∫ –∂–µ –±—É–¥–µ–º —Å—á–∏—Ç–∞—Ç—å –∏–∑–º–µ–Ω–µ–Ω–∏—è —Ç–µ–º–ø–µ—Ä–∞—Ç—É—Ä—ã –≤ –æ–ø—ã—Ç–µ –Ω–µ–±–æ–ª—å—à–∏–º 
$\frac{\Delta T}{T} \ll 1$. –¢–æ–≥–¥–∞ –ø–æ–ª–∞–≥–∞—è $\Delta H = 0$ –∏–∑ \ref{eq:entalpy}:
\begin{equation}\label{eq:mu}
    \mu_{\text{–¥-—Ç}} = \frac{\Delta T}{\Delta P} \approx -
    \frac{b - \frac{2a}{RT}}{C_{\text{P}}}
\end{equation}
–ò–∑ \ref{eq:mu} –≤–∏–¥–Ω–æ, —á—Ç–æ —Å—É—â–µ—Å—Ç–≤—É–µ—Ç —Ç–∞–∫–∞—è —Ç–µ–º–ø–µ—Ä–∞—Ç—É—Ä–∞ (\emph{—Ç–µ–º–ø–µ—Ä–∞—Ç—É—Ä–∞ –∏–Ω–≤–µ—Ä—Å–∏–∏}),
–ø—Ä–∏ –∫–æ—Ç–æ—Ä–æ–π –∑–Ω–∞–∫ —ç—Ñ—Ñ–µ–∫—Ç–∞ –º–µ–Ω—è–µ—Ç—Å—è:
\begin{equation}\label{eq:temperature}
    T_{\text{–∏–Ω–≤}} = \frac{2a}{Rb}
\end{equation}
